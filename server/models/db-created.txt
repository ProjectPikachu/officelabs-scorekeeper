CREATE TABLE Games
(
    GameId SERIAL PRIMARY KEY,
    WinnerID int,
    LoserID int,
    WinnerPoints int,
    LoserPoints int,
    WinnerConfirm BOOL,
    LoserConfirm BOOL,
    Foreign Key (WinnerID) REFERENCES Users(UserID),
    Foreign Key (LoserID) REFERENCES Users(UserID)
)


CREATE TABLE Users
(
    UserId SERIAL PRIMARY KEY,
    Username VARCHAR(100),
    Password VARCHAR(100),
    Cohort int,
    FullName VARCHAR(100),
    
)
